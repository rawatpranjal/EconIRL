%% ============================================================
\section{Context and Problem}
%% ============================================================

Maximum entropy IRL \citep{ziebart2010} recovers a reward function that rationalizes observed behavior, but the recovered reward entangles the true reward with the value function.
Formally, for any bounded function $h : \calS \to \R$, the shaped reward
\begin{equation}
\label{eq:shaping}
r'(s,a) \;=\; r(s,a) \;+\; \gamma\, \E_{s' \sim P(\cdot|s,a)}[h(s')] \;-\; h(s)
\end{equation}
produces the same optimal policy under the original transitions $P$ \citep{ng1999}.
If $P$ changes (new environment, counterfactual), the shaping term evaluates differently and the shaped reward no longer rationalizes the same behavior.
This matters in economics because counterfactual policy evaluation requires a reward that transfers across environments.

\citet{fu2018} propose AIRL to disentangle the transferable reward from the non-transferable value function.
The key idea: structure the discriminator so that one component converges to the true reward $r^*(s)$ and the other absorbs the value function $V^*(s)$.
At the adversarial optimum, the reward component is invariant to changes in dynamics.

\subsection{Maximum Causal Entropy Objective}

The agent maximizes entropy-regularized cumulative reward \citep{ziebart2010}:
\begin{equation}
\label{eq:objective}
\max_\pi \; \E_\pi\!\left[\sum_{t=0}^{\infty} \gamma^t \big(r(s_t, a_t) + \calH(\pi(\cdot \mid s_t))\big)\right].
\end{equation}
The optimal policy is the softmax over advantages:
\begin{equation}
\label{eq:softmax}
\pi^*(a \mid s) = \exp\!\big(A^*(s,a)\big) \Big/ \textstyle\sum_{a'} \exp\!\big(A^*(s,a')\big).
\end{equation}
Taking the log: $\log \pi^*(a \mid s) = A^*(s,a) = Q^*(s,a) - V^*(s)$.

\subsection{Discriminator Structure}

AIRL uses a structured discriminator that decomposes into reward and shaping:
\begin{align}
f_{\theta,\phi}(s,a,s') &= g_\theta(s) + \gamma\, h_\phi(s') - h_\phi(s), \label{eq:f}\\[3pt]
D_{\theta,\phi}(s,a,s') &= \frac{\exp(f_{\theta,\phi})}{\exp(f_{\theta,\phi}) + \pi_\omega(a \mid s)}. \label{eq:D}
\end{align}
The logit $f$ mirrors the Bellman relationship $Q(s,a) = r(s) + \gamma \E[V(s')]$, decomposed into $g_\theta \approx r$ and $h_\phi \approx V$.
Note that $g_\theta$ depends on $s$ only, not on $a$.
The discriminator classifies transitions as expert ($D \to 1$) or generated ($D \to 0$).

%% ============================================================
\section{Identification and Theorems}
%% ============================================================

\paragraph{Reward ambiguity (background).}
IRL rewards are identified only up to potential-based shaping~\eqref{eq:shaping} \citep{ng1999}.
\citet{cao2021} formalize this: the observed policy constrains $|\calS|(|\calA|-1)$ values but the reward has $|\calS| \cdot |\calA|$ unknowns, leaving $|\calS|$ free variables corresponding to the value function.

\paragraph{Decomposability (Fu et al., Definition B.1).}
An MDP is \emph{decomposable} if all states are linked: for every pair $(s, s')$ there exists a policy and a time $t$ such that $s'$ is reachable from $s$ with positive probability.
This is a standard ergodicity-like condition that rules out absorbing states.

\paragraph{Disentanglement (Fu et al., Theorem 1).}
If (a)~$g_\theta$ is state-only, (b)~the MDP is decomposable, and (c)~the adversarial training converges to the Nash equilibrium, then $g_\theta(s) = r^*(s) + c$ for a constant $c$.

The proof uses the fact that at the Nash equilibrium, $D = 1/2$ everywhere, so $f_{\theta,\phi} = \log \pi^* = A^*$.
Substituting the decomposition: $g_\theta(s) + \gamma h_\phi(s') - h_\phi(s) = r^*(s) + \gamma V^*(s') - V^*(s)$.
Since $g_\theta$ depends on $s$ only, taking the expectation over $s'$ gives $g_\theta(s) = r^*(s) + h_\phi(s) - V^*(s) + c_0$.
The decomposability condition then implies $h_\phi(s) - V^*(s)$ is constant across all states, so $g_\theta(s) = r^*(s) + c$.

\paragraph{Limitations.}
Theorem~1 requires state-only rewards.
If the true reward depends on actions ($r^*(s,a)$), as in DDC models where different actions have different payoffs, the disentanglement guarantee breaks.
\citet{lee2026} resolve this with anchor constraints (see AAIRL primer).
The decomposability condition excludes absorbing states, which appear in every DDC model with an exit option.

\section{Estimation Algorithm}
%% ============================================================

\begin{algorithm}[h!]
\caption{AIRL (Fu et al.\ 2018)}
\label{alg:airl}
\begin{algorithmic}[1]
\REQUIRE Expert demonstrations $\calD_E = \{\tau_1, \ldots, \tau_N\}$, discount $\gamma$
\ENSURE Disentangled reward $g_\theta(s)$

\STATE Initialize $\theta_0, \phi_0, \omega_0$

\FOR{iteration $= 1, 2, \ldots$}

\STATE \textbf{// Step 1: Sample from current policy}
\STATE $\calD_\pi \leftarrow$ roll out $\pi_\omega$ in environment to collect $M$ transitions

\STATE \textbf{// Step 2: Train discriminator}
\STATE Extract expert transitions: $\{(s, a, s')\} \sim \calD_E$
\STATE Extract policy transitions: $\{(s, a, s')\} \sim \calD_\pi$
\FOR{$j = 1, \ldots, J$ discriminator steps}
  \STATE $f_n \leftarrow g_\theta(s_n) + \gamma\, h_\phi(s'_n) - h_\phi(s_n)$ \hfill \textit{// Eq.~\eqref{eq:f}}
  \STATE $D_n \leftarrow \sigma(f_n - \log \pi_\omega(a_n \mid s_n))$ \hfill \textit{// Eq.~\eqref{eq:D}, $\sigma$ = sigmoid}
  \STATE $\calL \leftarrow -\E_{\calD_E}[\log D] - \E_{\calD_\pi}[\log(1-D)]$
  \STATE $\theta, \phi \leftarrow \theta, \phi - \alpha_D \nabla_{\theta,\phi} \calL$
\ENDFOR

\STATE \textbf{// Step 3: Update policy}
\STATE Compute reward signal for policy: $\tilde{r}(s,a,s') \leftarrow \log D - \log(1-D) = f - \log \pi_\omega$
\STATE Update $\omega$ via any policy optimization method (PPO in practice)

\STATE \textbf{// Step 4: Check convergence}
\IF{$D \approx 1/2$ on held-out data}
  \STATE \textbf{break}
\ENDIF

\ENDFOR
\RETURN $g_\theta$ \quad {\small (disentangled reward, transferable across dynamics)}
\end{algorithmic}
\end{algorithm}
\FloatBarrier

\paragraph{Key implementation details.}
The reward signal in Step~3 is $\tilde{r} = \log D - \log(1-D)$, which equals $f - \log \pi_\omega$ by construction of~\eqref{eq:D}.
This is the log-odds of the discriminator and serves as the advantage estimate.
\citet{fu2018} use PPO \citep{schulman2017} for the policy update (Step~3), though any policy optimization method works.
The discriminator and policy are trained alternately, with $J = 1$ discriminator step per policy step in their experiments.

\paragraph{Convergence.}
At the Nash equilibrium, $D = 1/2$ everywhere and $f_{\theta,\phi} = A^*$.
By Theorem~1, $g_\theta(s) = r^*(s) + c$.
The constant $c$ is economically irrelevant because it does not affect any policy or counterfactual prediction.

%% ============================================================
\section{Results from the Paper}
%% ============================================================

\citet{fu2018} test AIRL on continuous-control tasks from OpenAI Gym and transfer experiments.

\paragraph{Transfer experiment.}
They train AIRL on a PointMaze environment with one set of walls, then evaluate the recovered reward in a maze with different walls.
AIRL transfers successfully (the agent navigates the new maze), while GAIL and MaxEnt IRL fail because their recovered rewards are entangled with the original dynamics.
This is the paper's central result: disentanglement enables transfer.

\paragraph{Continuous control.}
On Ant, HalfCheetah, and Hopper, AIRL matches or exceeds GAIL in terms of policy performance (cumulative reward) while also recovering a reward function.
GAIL recovers a policy but not a reward.

\paragraph{State-only vs.\ state-action.}
When the true reward is state-only, AIRL with $g_\theta(s)$ recovers it accurately (Theorem~1).
When the true reward depends on actions, $g_\theta(s)$ absorbs the action-dependent variation into the shaping potential, and the recovered reward is less interpretable.
